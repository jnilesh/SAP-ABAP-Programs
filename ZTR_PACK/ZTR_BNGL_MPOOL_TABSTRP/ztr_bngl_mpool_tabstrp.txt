PROGRAM ZTR_BNGL_MPOOL_TABSTRP.
CONTROLS TABSTP TYPE TABSTRIP.

DATA: LV_NUM1 TYPE I,
      LV_NUM2 TYPE I,
      LV_N1 TYPE I,
      LV_N2 TYPE I,
      LV_RSLT TYPE I.
*&---------------------------------------------------------------------*
*&      Module  STATUS_0100  OUTPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE status_0100 OUTPUT.
  SET PF-STATUS 'ZTR_BNGL_PFS'.
*  SET TITLEBAR 'xxx'.
ENDMODULE.
*&---------------------------------------------------------------------*
*&      Module  USER_COMMAND_0100  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE user_command_0100 INPUT.
CASE SY-UCOMM.
  WHEN 'TAB1'.
  TABSTP-ACTIVETAB =  'TAB1'.

  WHEN 'TAB2'.
  TABSTP-ACTIVETAB =  'TAB2'.

  LV_RSLT = LV_NUM1 + LV_NUM2.
  WHEN 'EXIT'.
    LEAVE PROGRAM.

ENDCASE.



ENDMODULE.
*&---------------------------------------------------------------------*
*&      Module  USER_COMMAND_0101  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE user_command_0101 INPUT.

LV_N1  = LV_NUM1.
LV_N2  = LV_NUM2.

ENDMODULE.
*&---------------------------------------------------------------------*
*&      Module  USER_COMMAND_0102  INPUT
*&---------------------------------------------------------------------*
*       text
*----------------------------------------------------------------------*
MODULE user_command_0102 INPUT.


LV_RSLT = LV_NUM1 + LV_NUM2.


ENDMODULE.
