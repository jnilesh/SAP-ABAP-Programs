FORM SELECT_DATA_SEC .

              ENAME
              JOB
              MGR
              HIREDATE
              SAL
              DEPTNO
         INTO TABLE ITAB02 FROM
         Z160680_EMP
         WHERE
         DEPTNO = WA_ITAB01-DEPTNO.
