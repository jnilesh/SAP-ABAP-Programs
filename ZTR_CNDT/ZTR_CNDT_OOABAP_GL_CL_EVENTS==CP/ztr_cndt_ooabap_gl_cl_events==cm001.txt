  METHOD GET_MATERIAL.

    DATA LV_MATNR TYPE MATNR.
    SELECT SINGLE MATNR
       INTO LV_MATNR FROM MARA
  WHERE MATNR BETWEEN I_MATNR1 AND I_MATNR2.
    IF SY-SUBRC EQ 0.
            SELECT MATNR
           MBRSH
           MTART
           MEINS
           MATKL
           BRGEW
           NTGEW
         INTO TABLE IT_TAB
      FROM MARA
    WHERE MATNR BETWEEN I_MATNR1 AND I_MATNR2.
    ELSE.
            RAISE EVENT MAT_EVENT.
    ENDIF.

  ENDMETHOD.
